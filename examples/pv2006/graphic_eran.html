def add_cyclic_cols(df):
    for col in CYCLIC_COLS:
        df[f"{col}_sin"]  = np.sin((2*np.pi*df[col].astype(float))/max(df[col].astype(float)))
        df[f"{col}_cos"]  = np.cos((2*np.pi*df[col].astype(float))/max(df[col].astype(float)))
    return df
